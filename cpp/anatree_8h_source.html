232 static
233 node_ptr
234 deep_copy(const node_ptr &p)
235 {
236 node_ptr np = node::make_node();
237
238 // Case: NIL
239 // -> Leave node as-is
240 if (p->m_char == node::NIL) {
241 assert(p->m_children[false] == nullptr && p->m_children[true] == nullptr);
242 }
243
244 // Case: not NIL
245 // -> Obtain recursively deep copy of children and copy over 'm_char'
246 else {
247 np->m_char = p->m_char;
248 np->m_children[false] = deep_copy(p->m_children[false]);
249 np->m_children[true] = deep_copy(p->m_children[true]);
250 }
251
252 // Copy over the set of words stored within this node.
253 np->m_words = p->m_words;
254 return std::move(np);
255 }

289private:
290 node_ptr
291 insert__rec(node_ptr &&p,
292 const T &w,
293 typename T::iterator curr,
294 const typename T::iterator end)
295 {
296 assert(p != nullptr);
297
298 // Case: Iterator done
299 // -> Insert word
300 if (curr == end) {
301 if (!p->m_words.contains(w)) {
302 m_size++;
303 p->m_words.insert(w);
304 }
305 return p;
306 }
307
308 // Case: NIL
309 // -> Turn into non-NIL node
310 if (p->m_char == node::NIL) {
311 assert(p->m_children[false] == nullptr && p->m_children[true] == nullptr);
312 p->init(*curr);
313 m_tree_size += 2; // <- NIL 'false' and 'true' children
314 p->m_children[true] = insert__rec(std::move(p->m_children[true]), w, ++curr, end);
315 return p;
316 }
317
318 // Case: Iterator behind
319 // -> Insert new node in-between
320 if (m_char_comp(*curr, p->m_char)) {
321 // Hand over ownership of 'p' to a new node 'np'. The original node 'p' is
322 // now available as its 'false' child.
323 node_ptr np = node::make_node(*curr, std::move(p), node::make_node());
324 np->m_words = std::move(np->m_children[false]->m_words);
325 np->m_children[false]->m_words = Set();
326 m_tree_size += 2; // <- new node and its NIL 'false' child
327 np->m_children[true] = insert__rec(std::move(np->m_children[true]), w, ++curr, end);
328 return std::move(np);
329 }
330
331 // Case: Iterator ahead
332 // -> Follow 'false' child
333 if (m_char_comp(p->m_char, *curr)) {
334 p->m_children[false] = insert__rec(std::move(p->m_children[false]), w, curr, end);
335 return p;
336 }
337
338 // Case: Iterator and node matches
339 // -> Follow 'true' child
340 p->m_children[true] = insert__rec(std::move(p->m_children[true]), w, ++curr, end);
341 return p;
342 }

385private:
386 Map
387 keys__rec(const node_ptr &p) const
388 {
389 // Case: Leaf of Tree
390 // -> Add a word, if any.
391 if (p->m_char == node::NIL) {
392 Map ret;
393 if (p->m_words.size() > 0) {
394 ret[{}] = *p->m_words.begin();
395 }
396 return ret;
397 }
398
399 // Case: Internal Node
400 // -> Recurse for words with and without this character
401 auto rec_true = keys__rec(p->m_children[true]);
402 auto rec_false = keys__rec(p->m_children[false]);
403
404 Map ret;
405
406 // -> Copy over words excluding current node's character (except for ones
407 // that are superseeded)
408 for (const auto [rec_false_k, rec_false_v] : rec_false) {
409 // Check whether the word is superseeded by any word in 'rec_true'.
410 // - (a) Simple O(1) case
411 if (rec_true.contains(rec_false_v)) { continue; }
412
413 // - (b) Expensive O(T) case
414 bool superseeded = false;
415 for (const auto [rec_true_k, _] : rec_true) {
416 int false_idx = rec_false_k.length() - 1;
417 int true_idx = rec_true_k.length() - 1;
418
419 while (0 <= false_idx && 0 <= true_idx) {
420 const value_type false_char = rec_false_k[false_idx];
421 const value_type true_char = rec_true_k[true_idx];
422
423 if (m_char_comp(true_char, false_char)) {
424 true_idx--;
425 continue;
426 } else if (false_char == true_char) {
427 false_idx--;
428 true_idx--;
429 continue;
430 } else { // False includes character not in True path.
431 break;
432 }
433 }
434 if (false_idx < 0 && true_idx < 0) {
435 superseeded = true;
436 break;
437 }
438 }
439 if (superseeded) { continue; }
440
441 // Add word
442 T curr_key(rec_false_k);
443 curr_key.push_back(p->m_char);
444
445 assert(!ret.contains(curr_key));
446 ret[curr_key] = rec_false_v;
447 }
448 rec_false.clear(); // <-- Save on memory
449
450 // -> Copy over words including current node's character.
451 for (const auto [rec_k, rec_v] : rec_true) {
452 T curr_key(rec_k);
453 curr_key.push_back(p->m_char);
454
455 ret[curr_key] = rec_v;
456 }
457 // rec_true.clear(); // <-- Save on memory (unecessary)
458
459 return ret;
460 }

477private:
478 void
479 keys__rec(const size_t word_length,
480 const node_ptr &p,
481 const size_t true_edges,
482 Set &res) const
483 {
484 assert(true_edges <= word_length);
485
486 // Case: Found word of 'word_length'
487 // -> Search succesful (no need to keep on searching deeper)
488 if (word_length == true_edges) {
489 if (p->m_words.size() > 0) {
490 res.insert(*p->m_words.begin());
491 }
492 return;
493 }
494
495 // Case: Tree stopped early
496 // -> Abandon subtree
497 if (p->m_char == node::NIL) {
498 return;
499 }
500
501 // Case: Missing characters
502 // -> Merge recursively from false and true subtrees
503 keys__rec(word_length, p->m_children[true], true_edges+1, res);
504 keys__rec(word_length, p->m_children[false], true_edges, res);
505 }

551private:
552 void
553 subanagrams_of__rec(const node_ptr &p,
554 typename T::iterator curr,
555 const typename T::iterator end,
556 Set &res) const
557 {
558 res.insert(p->m_words.begin(), p->m_words.end());
559
560 // Case: Anatree is done
561 // -> Stop
562 if (p->m_char == node::NIL) {
563 return;
564 }
565
566 // Case: Iterator behind
567 // -> Skip missing characters
568 while (curr != end && m_char_comp(*curr, p->m_char)) { ++curr; }
569
570 // Case: Iterator done
571 // -> Stop
572 if (curr == end) {
573 return;
574 }
575
576 // Case: Iterator ahead
577 // -> Follow 'false' child
578 if (m_char_comp(p->m_char, *curr)) {
579 subanagrams_of__rec(p->m_children[false], curr, end, res);
580 return;
581 }
582
583 // Case: Iterator and node matches
584 // -> Follow both children, merge results and add words on current node
585 ++curr;
586 subanagrams_of__rec(p->m_children[false], curr, end, res);
587 subanagrams_of__rec(p->m_children[true], curr, end, res);
588 }

654 const node_ptr& find_node__rec(const node_ptr &p,
655 typename T::iterator curr,
656 const typename T::iterator end) const
657 {
658 // Case: Iterator is done
659 if (curr == end) {
660 return p;
661 }
662
663 // Case: Iterator behind or tree is done
664 // -> No words with all letters exist, return Ã .
665 if (m_char_comp(*curr, p->m_char) || p->m_char == node::NIL) {
666 return m_null;
667 }
668
669 // Case: Iterator ahead
670 // -> Follow 'false' child
671 if (m_char_comp(p->m_char, *curr)) {
672 return find_node__rec(p->m_children[false], curr, end);
673 }
674
675 // Case: Iterator and node matches
676 // -> Follow 'true' child
677 return find_node__rec(p->m_children[true], ++curr, end);
678 }
